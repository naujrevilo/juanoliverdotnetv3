export async function getStaticPaths() {
  let lastModified = "";
  if (Astro.server) {
    const res = await fetch(
      `${
        Astro.site.origin || ""
      }/api/last-modified?file=src/pages/privacidad.mdx`
    );
    const data = await res.json();
    lastModified = data.lastModified || "";
  }
  return [{ params: {}, props: { lastModified } }];
}
